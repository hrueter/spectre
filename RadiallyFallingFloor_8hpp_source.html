30 /// \ingroup VariableFixingGroup
31 /// \brief Applies a pressure and density floor dependent on the distance
32 /// to the origin.
33 ///
34 /// Applies the floors:
35 /// \f$\rho(r) \geq \rho_{\mathrm{fl}}(r) = 10^{-5}r^{-3/2}\f$
36 /// and \f$P(r) \geq P_{\mathrm{fl}}(r) = \frac{1}{3} \times 10^{-7}r^{-5/2}\f$
37 /// when \f$ r > \f$`radius_at_which_to_begin_applying_floor`.
38 /// These bounds are described in Porth et al.'s ["The Black Hole Accretion
39 /// Code"](https://arxiv.org/pdf/1611.09720.pdf).
40 template <size_t Dim, typename Density, typename Pressure>
41 struct RadiallyFallingFloor {
42  using return_tags = tmpl::list<Density, Pressure>;
43  using argument_tags = tmpl::list<::Tags::Coordinates<Dim, Frame::Inertial>,
44  CacheTags::MaskRadius>;
45 
46  static void apply(
47  const gsl::not_null<Scalar<DataVector>*> density,
48  const gsl::not_null<Scalar<DataVector>*> pressure,
49  const tnsr::I<DataVector, Dim, Frame::Inertial>& coords,
50  const double radius_at_which_to_begin_applying_floor) noexcept {
51  const auto radii = magnitude(coords);
52  for (size_t i = 0; i < density->get().size(); i++) {
53  if (UNLIKELY(radii.get()[i] < radius_at_which_to_begin_applying_floor)) {
54  continue;
55  }
56  const double& radius = radii.get()[i];
57  const double radius_to_the_three_halves_power = sqrt(radius) * radius;
58 
59  pressure->get()[i] =
60  std::max(pressure->get()[i],
61  (1.e-7 / 3.) / (radius * radius_to_the_three_halves_power));
62  density->get()[i] =
63  std::max(density->get()[i], 1.e-5 / radius_to_the_three_halves_power);
64  }
65  }
66 };
67 } // namespace VariableFixing
