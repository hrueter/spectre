147 template <typename Component>
148 template <size_t... Is>
149 void MockLocalAlgorithm<Component>::next_action(
150  std::index_sequence<Is...> /*meta*/) noexcept {
151  auto& const_global_cache = *const_global_cache_;
152  if (UNLIKELY(performing_action_)) {
153  ERROR(
154  "Cannot call an Action while already calling an Action on the same "
155  "MockLocalAlgorithm (an element of a parallel component array, or a "
156  "parallel component singleton).");
157  }
158  // Keep track of if we already evaluated an action since we want `next_action`
159  // to only evaluate one per call.
160  bool already_did_an_action = false;
161  const auto helper = [
162  this, &array_index = array_index_, &inboxes = *inboxes_,
163  &const_global_cache, &already_did_an_action
164  ](auto iteration) noexcept {
165  constexpr size_t iter = decltype(iteration)::value;
166  using this_action = tmpl::at_c<actions_list, iter>;
167  using this_databox =
168  tmpl::at_c<databox_types,
169  iter == 0 ? tmpl::size<databox_types>::value - 1 : iter>;
170  if (already_did_an_action or algorithm_step_ != iter) {
171  return;
172  }
173 
174  this_databox* box_ptr{};
175  try {
176  box_ptr = &boost::get<this_databox>(box_);
177  } catch (std::exception& e) {
178  ERROR(
179  "\nFailed to retrieve Databox in take_next_action:\nCaught "
180  "exception: '"
181  << e.what() << "'\nDataBox type: '"
182  << pretty_type::get_name<this_databox>() << "'\nIteration: " << iter
183  << "\nAction: '" << pretty_type::get_name<this_action>()
184  << "'\nBoost::Variant id: " << box_.which()
185  << "\nBoost::Variant type is: '" << type_of_current_state(box_)
186  << "'\n\n");
187  }
188  this_databox& box = *box_ptr;
189 
190  const auto check_if_ready = make_overloader(
191  [&box, &array_index, &const_global_cache, &inboxes](
192  std::true_type /*has_is_ready*/, auto t) {
193  return decltype(t)::is_ready(
194  cpp17::as_const(box), cpp17::as_const(inboxes),
195  const_global_cache, cpp17::as_const(array_index));
196  },
197  [](std::false_type /*has_is_ready*/, auto) { return true; });
198  if (not check_if_ready(Parallel::Algorithm_detail::is_is_ready_callable_t<
199  this_action, this_databox,
200  tuples::TaggedTupleTypelist<inbox_tags_list>,
201  Parallel::ConstGlobalCache<metavariables>,
202  typename Component::index>{},
203  this_action{})) {
204  ERROR("Tried to invoke the action '"
205  << pretty_type::get_name<this_action>()
206  << "' but have not received all the "
207  "necessary data.");
208  }
209  performing_action_ = true;
210  algorithm_step_++;
211  constexpr Component const* const component_ptr = nullptr;
212  make_overloader(
213  [ this, &array_index, component_ptr, &const_global_cache, &inboxes ](
214  auto& my_box, std::integral_constant<size_t, 1> /*meta*/)
215  SPECTRE_JUST_ALWAYS_INLINE noexcept {
216  std::tie(box_) = this_action::apply(
217  my_box, inboxes, const_global_cache,
218  cpp17::as_const(array_index), actions_list{}, component_ptr);
219  },
220  [ this, &array_index, component_ptr, &const_global_cache, &inboxes ](
221  auto& my_box, std::integral_constant<size_t, 2> /*meta*/)
222  SPECTRE_JUST_ALWAYS_INLINE noexcept {
223  std::tie(box_, terminate_) = this_action::apply(
224  my_box, inboxes, const_global_cache,
225  cpp17::as_const(array_index), actions_list{}, component_ptr);
226  },
227  [ this, &array_index, component_ptr, &const_global_cache, &inboxes ](
228  auto& my_box, std::integral_constant<size_t, 3> /*meta*/)
229  SPECTRE_JUST_ALWAYS_INLINE noexcept {
230  std::tie(box_, terminate_, algorithm_step_) = this_action::apply(
231  my_box, inboxes, const_global_cache,
232  cpp17::as_const(array_index), actions_list{}, component_ptr);
233  })(
234  box,
235  typename std::tuple_size<decltype(this_action::apply(
236  box, inboxes, const_global_cache, cpp17::as_const(array_index),
237  actions_list{}, component_ptr))>::type{});
238  performing_action_ = false;
239  already_did_an_action = true;
240 
241  // Wrap counter if necessary
242  if (algorithm_step_ >= tmpl::size<actions_list>::value) {
243  algorithm_step_ = 0;
244  }
245  };
246  // Silence compiler warning when there are no Actions.
247  (void)helper;
248  EXPAND_PACK_LEFT_TO_RIGHT(helper(std::integral_constant<size_t, Is>{}));
249 }

251 template <typename Component>
252 template <size_t... Is>
253 bool MockLocalAlgorithm<Component>::is_ready(
254  std::index_sequence<Is...> /*meta*/) noexcept {
255  bool next_action_is_ready = false;
256  const auto helper = [
257  this, &array_index = array_index_, &inboxes = *inboxes_,
258  &const_global_cache = const_global_cache_, &next_action_is_ready
259  ](auto iteration) noexcept {
260  constexpr size_t iter = decltype(iteration)::value;
261  using this_action = tmpl::at_c<actions_list, iter>;
262  using this_databox =
263  tmpl::at_c<databox_types,
264  iter == 0 ? tmpl::size<databox_types>::value - 1 : iter>;
265  if (iter != algorithm_step_) {
266  return;
267  }
268 
269  this_databox* box_ptr{};
270  try {
271  box_ptr = &boost::get<this_databox>(box_);
272  } catch (std::exception& e) {
273  ERROR(
274  "\nFailed to retrieve Databox in take_next_action:\nCaught "
275  "exception: '"
276  << e.what() << "'\nDataBox type: '"
277  << pretty_type::get_name<this_databox>() << "'\nIteration: " << iter
278  << "\nAction: '" << pretty_type::get_name<this_action>()
279  << "'\nBoost::Variant id: " << box_.which()
280  << "\nBoost::Variant type is: '" << type_of_current_state(box_)
281  << "'\n\n");
282  }
283  this_databox& box = *box_ptr;
284 
285  const auto check_if_ready = make_overloader(
286  [&box, &array_index, &const_global_cache, &inboxes](
287  std::true_type /*has_is_ready*/, auto t) {
288  return decltype(t)::is_ready(
289  cpp17::as_const(box), cpp17::as_const(inboxes),
290  *const_global_cache, cpp17::as_const(array_index));
291  },
292  [](std::false_type /*has_is_ready*/, auto) { return true; });
293 
294  next_action_is_ready =
295  check_if_ready(Parallel::Algorithm_detail::is_is_ready_callable_t<
296  this_action, this_databox,
297  tuples::TaggedTupleTypelist<inbox_tags_list>,
298  Parallel::ConstGlobalCache<metavariables>,
299  typename Component::index>{},
300  this_action{});
301  };
302  // Silence compiler warning when there are no Actions.
303  (void)helper;
304  EXPAND_PACK_LEFT_TO_RIGHT(helper(std::integral_constant<size_t, Is>{}));
305  return next_action_is_ready;
306 }
