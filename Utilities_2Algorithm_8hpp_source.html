85 /*!
86  * \ingroup UtilitiesGroup
87  * Reimplementation of std::next_permutation that is constexpr,
88  * for a generic comparator; taken from the LLVM source at
89  * https://github.com/llvm-mirror/libcxx/blob/master/include/algorithm
90  */
91 template <class Compare, class BidirectionalIterator>
92 constexpr bool next_permutation(BidirectionalIterator first,
93  BidirectionalIterator last, Compare comp) {
94  BidirectionalIterator i = last;
95  if (first == last || first == --i) {
96  return false;
97  }
98  while (true) {
99  BidirectionalIterator ip1 = i;
100  if (comp(*--i, *ip1)) {
101  BidirectionalIterator j = last;
102  while (!comp(*i, *--j)) {
103  }
104  cpp20::swap(*i, *j);
105  cpp20::reverse(ip1, last);
106  return true;
107  }
108  if (i == first) {
109  cpp20::reverse(first, last);
110  return false;
111  }
112  }
113 }
